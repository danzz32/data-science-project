Introdução

Este notebook tem como objetivo documentar o perfil de qualidade dos dados brutos de acidentes da PRF (camada raw/). O diagnóstico guiará as decisões de tratamento que serão implementadas no script transform.py. A análise está dividida nas quatro dimensões de Batini et al. (2009): Completude, Consistência, Acurácia e Temporalidade.

1. Dimensão: Completude

Objetivo: Identificar campos com valores ausentes (nulos), a proporção de falta e se existe algum padrão.

Code 
import duckdb
import os
import pandas as pd
from pathlib import Path

print("🎯 Iniciando Diagnóstico de Completude e Repetição de Identificadores...\n")

# 1. Configuração de caminhos dinâmicos
PROJECT_ROOT = Path(os.getcwd()).resolve().parent
RAW_DIR = PROJECT_ROOT / "data" / "raw"

arquivos_csv = list(RAW_DIR.glob("prf_191_*.csv"))

if arquivos_csv:
    PATH_CSV_BRUTO = arquivos_csv[-1]
    con = duckdb.connect(database=':memory:')
    
    # Cria a view mestre no DuckDB para otimizar as queries
    con.execute(f"""
        CREATE OR REPLACE VIEW v_prf_bruto AS 
        SELECT * FROM read_csv_auto('{PATH_CSV_BRUTO}', delim=';', encoding='latin-1', all_varchar=True);
    """)
    
    # Detecta qual é a coluna de identificação de pessoas disponível
    colunas_disponiveis = [col[0] for col in con.execute("DESCRIBE v_prf_bruto;").fetchall()]
    col_pessoa = 'pesid' if 'pesid' in colunas_disponiveis else ('id_envolvido' if 'id_envolvido' in colunas_disponiveis else None)

    # =====================================================================
    # PARTE 1: CONTADOR DE REPETIÇÃO DOS IDENTIFICADORES (Granularidade)
    # =====================================================================
    print("🔄 [Análise de Repetição] Avaliando a frequência de chaves no dataset geral:")
    
    query_repeticao = """
        WITH freq_acidentes AS (
            SELECT id, COUNT(*) as qtd FROM v_prf_bruto GROUP BY id
        ),
        freq_veiculos AS (
            SELECT id_veiculo, COUNT(*) as qtd FROM v_prf_bruto WHERE id_veiculo IS NOT NULL AND id_veiculo != '0' GROUP BY id_veiculo
        )
    """
    
    if col_pessoa:
        query_repeticao += f""",
        freq_pessoas AS (
            SELECT {col_pessoa}, COUNT(*) as qtd FROM v_prf_bruto GROUP BY {col_pessoa}
        )
        """
        
    query_repeticao += """
        SELECT 
            (SELECT AVG(qtd) FROM freq_acidentes) as avg_acidente,
            (SELECT MAX(qtd) FROM freq_acidentes) as max_acidente,
            (SELECT AVG(qtd) FROM freq_veiculos) as avg_veiculo,
            (SELECT MAX(qtd) FROM freq_veiculos) as max_veiculo
    """
    
    if col_pessoa:
        query_repeticao += """,
            (SELECT AVG(qtd) FROM freq_pessoas) as avg_pessoa,
            (SELECT MAX(qtd) FROM freq_pessoas) as max_pessoa;
        """
    else:
        query_repeticao += """,
            1.0 as avg_pessoa,
            1 as max_pessoa;
        """
        
    frequencia_ids = con.execute(query_repeticao).df()
    
    # Valores extraídos para variáveis locais para prevenir qualquer quebra sintática de f-string longa
    avg_ac, max_ac = frequencia_ids['avg_acidente'].values[0], frequencia_ids['max_acidente'].values[0]
    avg_vc, max_vc = frequencia_ids['avg_veiculo'].values[0], frequencia_ids['max_veiculo'].values[0]
    
    print(f"   • ID do Acidente : Média {avg_ac:.2f} linhas/ocorrência (Máx: {max_ac}).")
    print(f"   • ID do Veículo  : Média {avg_vc:.2f} linhas/veículo (Máx: {max_vc}).")
    
    if col_pessoa:
        avg_ps, max_ps = frequencia_ids['avg_pessoa'].values[0], frequencia_ids['max_pessoa'].values[0]
        print(f"   • ID da Pessoa ('{col_pessoa}'): Média {avg_ps:.2f} linhas/indivíduo (Máx: {max_ps}).")

    # =====================================================================
    # PARTE 2: VISÃO GERAL DA COMPLETUDE (Percentual de Nulos por Coluna)
    # =====================================================================
    print("\n📊 [Completude] Calculando a taxa de preenchimento e ausência por coluna...")
    
    # Captura o total de linhas para o cálculo percentual
    total_linhas = con.execute("SELECT COUNT(*) FROM v_prf_bruto;").fetchone()[0]
    print(f"   • Volume de amostragem: {total_linhas:,} linhas analisadas.".replace(',', '.'))
    
    # Construção dinâmica da query de contagem de nulos para cada coluna (detecta vazios, nulos e 'NA')
    select_nulos = ", ".join([f"COUNT(CASE WHEN {col} IS NULL OR {col} = '' OR {col} = 'NA' THEN 1 END) as {col}" for col in colunas_disponiveis])
    df_contagem_nulos = con.execute(f"SELECT {select_nulos} FROM v_prf_bruto;").df()
    
    # Transpõe o resultado para estruturar o DataFrame de ranking de completude
    df_nulos = pd.DataFrame({
        'Total_Nulos': df_contagem_nulos.iloc[0].values,
        'Percentual_Nulos (%)': (df_contagem_nulos.iloc[0].values / total_linhas) * 100
    }, index=colunas_disponiveis).sort_values(by='Total_Nulos', ascending=False)
    
    # Cópia formatada para exibição no Quarto/Notebook (Garantindo o sinal de %)
    df_nulos_view = df_nulos.copy()
    df_nulos_view['Percentual_Nulos (%)'] = df_nulos_view['Percentual_Nulos (%)'].map('{:.2f}%'.format)
    df_nulos_view['Total_Nulos'] = df_nulos_view['Total_Nulos'].map('{:,}'.format).str.replace(',', '.')
    
    print("\n📋 Tabela Completa de Completude (Todas as Colunas do Dataset Bruto):")
    display(df_nulos_view)

    # =====================================================================
    # PARTE 3: INVESTIGAÇÃO DE PADRÕES (O Requisito do PDF)
    # =====================================================================
    # Filtra colunas com nulos reais excluindo os IDs de controle para encontrar a variável de negócio mais crítica
    colunas_com_nulos = df_nulos[(df_nulos['Total_Nulos'] > 0) & (~df_nulos.index.isin(['id_veiculo', col_pessoa, 'id']))]
    coluna_critica = colunas_com_nulos.index[0] if not colunas_com_nulos.empty else None

    if coluna_critica:
        print(f"\n🔍 [Investigação] Analisando padrões de ausência para a coluna mais crítica: '{coluna_critica}'")
        
        # A. Padrão Temporal: Distribuição por Ano
        print("\n📅 Distribuição de Ausências por Ano (%):")
        padrao_temporal = con.execute(f"""
            SELECT 
                CASE 
                    WHEN LENGTH(data_inversa) >= 10 THEN SUBSTR(data_inversa, 1, 4)
                    ELSE data_inversa 
                END as ano_analise,
                ROUND(COUNT(CASE WHEN {coluna_critica} IS NULL OR {coluna_critica} = '' OR {coluna_critica} = 'NA' THEN 1 END) * 100.0 / COUNT(*), 2) as percentual_nulo
            FROM v_prf_bruto
            GROUP BY ano_analise
            ORDER BY ano_analise;
        """).df()
        
        # Formata a coluna com o sinal de % antes de exibir
        padrao_temporal['percentual_nulo'] = padrao_temporal['percentual_nulo'].map('{:.2f}%'.format)
        display(padrao_temporal.set_index('ano_analise'))
        
        # B. Padrão Categórico: Distribuição por Estado (UF)
        if 'uf' in colunas_disponiveis:
            print("\n📍 Distribuição de Ausências por UF (%):")
            padrao_uf = con.execute(f"""
                SELECT 
                    uf,
                    ROUND(COUNT(CASE WHEN {coluna_critica} IS NULL OR {coluna_critica} = '' OR {coluna_critica} = 'NA' THEN 1 END) * 100.0 / COUNT(*), 2) as percentual_nulo
                FROM v_prf_bruto
                GROUP BY uf
                ORDER BY percentual_nulo DESC;
            """).df()
            
            # Formata a coluna com o sinal de % antes de exibir
            padrao_uf['percentual_nulo'] = padrao_uf['percentual_nulo'].map('{:.2f}%'.format)
            display(padrao_uf.set_index('uf'))
            
    else:
        print("\n✨ Excelente! Nenhuma coluna de negócio possui valores nulos nativos no dataset bruto.")

else:
    print(f"❌ Erro: Nenhum arquivo bruto foi localizado em {RAW_DIR}")

Objetivo: Verificar contradições internas na base de dados.

import duckdb import os import pandas as pd from pathlib import Path

# Detecta as colunas disponíveis na base
colunas_disponiveis = [col[0] for col in con.execute("DESCRIBE v_prf_bruto;").fetchall()]

# Mapeia dinamicamente os nomes exatos das colunas numéricas de gravidade
col_mortos = 'mortos' if 'mortos' in colunas_disponiveis else None
col_f_leves = 'feridos_leves' if 'feridos_leves' in colunas_disponiveis else None
col_f_graves = 'feridos_graves' if 'feridos_graves' in colunas_disponiveis else None
col_ilesos = 'ilesos' if 'ilesos' in colunas_disponiveis else None
col_ignorados = 'ignorados' if 'ignorados' in colunas_disponiveis else None
col_pessoas = 'pessoas' if 'pessoas' in colunas_disponiveis else None
col_feridos = 'feridos' if 'feridos' in colunas_disponiveis else None

# =====================================================================
# PARTE 1: CONSTRUÇÃO DA VIEW DE SANITIZAÇÃO (Casting Defensivo)
# =====================================================================
def cast_inteiro(coluna):
    if coluna and coluna in colunas_disponiveis:
        return f"COALESCE(TRY_CAST({coluna} AS INTEGER), 0)"
    return "0"

con.execute(f"""
    CREATE OR REPLACE VIEW v_consistencia_sanitizada AS
    SELECT 
        *,
        LOWER(CAST(COALESCE(classificacao_acidente, '') AS VARCHAR)) as classe_clean,
        TRY_CAST(latitude AS DOUBLE) as lat_num,
        TRY_CAST(longitude AS DOUBLE) as lon_num,
        {cast_inteiro(col_mortos)} as n_mortos,
        {cast_inteiro(col_f_leves)} as n_f_leves,
        {cast_inteiro(col_f_graves)} as n_f_graves,
        {cast_inteiro(col_ilesos)} as n_ilesos,
        {cast_inteiro(col_ignorados)} as n_ignorados,
        {cast_inteiro(col_pessoas)} as n_pessoas,
        {cast_inteiro(col_feridos)} as n_feridos
    FROM v_prf_bruto;
""")

# =====================================================================
# PARTE 2: EXECUÇÃO DAS REGRAS DE CONSISTÊNCIA
# =====================================================================
print("📊 [Resultados da Auditoria] Incoerências detectadas na base:")

# Regra 1: Valores numéricos negativos
r1 = con.execute("""
    SELECT COUNT(*) FROM v_consistencia_sanitizada 
    WHERE n_mortos < 0 OR n_f_leves < 0 OR n_f_graves < 0 OR n_ilesos < 0 
       OR n_ignorados < 0 OR n_pessoas < 0;
""").fetchone()[0]
print(f"   ❌ Regra 1: Registros com valores numéricos negativos: {r1:,}".replace(',', '.'))

# Regra 2: Vítimas sem associação a um ID de Veículo Válido
r2 = con.execute("""
    SELECT COUNT(*) FROM v_consistencia_sanitizada 
    WHERE (id_veiculo IS NULL OR id_veiculo = '' OR id_veiculo = '0' OR id_veiculo = 'NA') 
      AND (n_mortos > 0 OR n_f_leves > 0 OR n_f_graves > 0);
""").fetchone()[0]
print(f"   ❌ Regra 2: Vítimas registradas sem associação a um veículo (id_veiculo ausente/zerado): {r2:,}".replace(',', '.'))

# Regra 3: Consistência na Matemática de Feridos (Leves + Graves != Total)
if col_feridos:
    r3 = con.execute("""
        SELECT COUNT(*) FROM v_consistencia_sanitizada 
        WHERE (n_f_leves + n_f_graves) != n_feridos;
    """).fetchone()[0]
    print(f"   ❌ Regra 3: Incoerência na soma de feridos (Leves + Graves != Total): {r3:,}".replace(',', '.'))
else:
    print("   🔹 Regra 3: Ignorada (Coluna macro 'feridos' não existe neste layout).")

# Regra 4: Consistência na Matemática Total de Pessoas (Soma dos componentes != Campo Pessoas)
r4 = con.execute("""
    SELECT COUNT(*) FROM v_consistencia_sanitizada 
    WHERE (n_mortos + n_f_leves + n_f_graves + n_ilesos + n_ignorados) != n_pessoas;
""").fetchone()[0]
print(f"   ❌ Regra 4: Incoerência na soma total de pessoas (Categorias != Campo Pessoas): {r4:,}".replace(',', '.'))

# Regra 5: Contradição Textual - Classificação diz "Com Vítimas Mortas" mas contador de mortos é zero
r5 = con.execute("""
    SELECT COUNT(*) FROM v_consistencia_sanitizada

""").fetchone()[0]

# Regra 6: Contradição Textual - Classificação diz "Sem Vítimas" mas registra feridos/mortos
r6 = con.execute("""
    SELECT COUNT(*) FROM v_consistencia_sanitizada 
    WHERE (classe_clean LIKE '%sem vít%' OR classe_clean LIKE '%sem vit%') 
      AND (n_mortos > 0 OR n_f_leves > 0 OR n_f_graves > 0);
""").fetchone()[0]
print(f"   ❌ Regra 6: Contradição em 'Sem Vítimas' mas registrando feridos/mortos: {r6:,}".replace(',', '.'))

# Regra 7: Consistência Geográfica (Coordenadas zeradas ou fora do padrão do território brasileiro)
r7 = con.execute("""
    SELECT COUNT(*) FROM v_consistencia_sanitizada 
    WHERE lat_num = 0.0 OR lon_num = 0.0 
       OR lat_num > 6.0 OR lat_num < -34.0 
       OR lon_num > -34.0 OR lon_num < -74.0;
""").fetchone()[0]
print(f"   ❌ Regra 7: Registros com coordenadas de GPS inválidas ou fora do Brasil: {r7:,}".replace(',', '.'))

        else:
            axes[idx_ax].text(0.5, 0.5, 'Sem dados válidos', ha='center', va='center')

    plt.tight_layout()
    plt.show()

        CREATE OR REPLACE VIEW v_prf_bruto AS

    """)

    if 'data_inversa' in colunas_disponiveis:

        # PARTE 1: VIEW COM PARSING DE DATA DEFENSIVO
        # =====================================================================
        # O TRY_CAST tenta converter formatos ISO (YYYY-MM-DD) ou BR (DD/MM/YYYY) automaticamente
        con.execute("""
            CREATE OR REPLACE VIEW v_temporalidade_validada AS
            SELECT 
                *,
                COALESCE(TRY_CAST(data_inversa AS DATE), strptime(data_inversa, '%d/%m/%Y')::DATE) as data_validada
            FROM v_prf_bruto;
        """)

        # Regra 1: Registros com datas corrompidas ou inválidas
        r1 = con.execute("SELECT COUNT(*) FROM v_temporalidade_validada WHERE data_validada IS NULL;").fetchone()[0]
        print(f"   ❌ Regra 1: Registros com datas corrompidas ou inválidas: {r1}")

        # Verifica se há dados válidos para continuar as análises temporais
        total_linhas = con.execute("SELECT COUNT(*) FROM v_temporalidade_validada;").fetchone()[0]
        
        if r1 < total_linhas:
            # Regra 2: Cobertura da Série Histórica
            datas_limite = con.execute("SELECT MIN(data_validada), MAX(data_validada) FROM v_temporalidade_validada;").fetchone()
            data_min, data_max = datas_limite[0], datas_limite[1]
            
            print(f"   ⏱️  Regra 2: Cobertura da Série Histórica:")
            print(f"      - Ponto inicial dos dados: {data_min.strftime('%d/%m/%Y')}")
            print(f"      - Ponto final dos dados:   {data_max.strftime('%d/%m/%Y')}")
            
            # Regra 3: Check de Viagem no Tempo (Datas além do dia de hoje)
            # Nota: O DuckDB executa a comparação contra a data atual do sistema (CURRENT_DATE)
            r3_df = con.execute("""
                SELECT data_inversa, COUNT(*) as qtd 
                FROM v_temporalidade_validada 
                WHERE data_validada > CURRENT_DATE
                GROUP BY data_inversa;
            """).df()
            
            print(f"   ❌ Regra 3: Registros com datas futuras aberrantes: {r3_df['qtd'].sum()}")
            if len(r3_df) > 0:
                print(f"      ⚠️ Amostra de datas futuras detectadas: {r3_df['data_inversa'].unique()[:5]}")

            # Regra 4: Investigação de Lacunas (Gaps Críticos por Dias Zerados)
            # Geramos uma série temporal contínua no SQL e cruzamos com os dias reais do dataset
            r4 = con.execute(f"""
                WITH calendario AS (
                    SELECT CAST(generate_series AS DATE) as dia_calendario
                    FROM generate_series(CAST('{data_min}' AS DATE), CAST('{data_max}' AS DATE), INTERVAL '1 DAY')
                ),
                dados_diarios AS (
                    SELECT data_validada, COUNT(*) as qtd 
                    FROM v_temporalidade_validada 
                    WHERE data_validada IS NOT NULL
                    GROUP BY data_validada
                )
                SELECT COUNT(*) 
                FROM calendario c
                LEFT JOIN dados_diarios d ON c.dia_calendario = d.data_validada
                WHERE d.data_validada IS NULL;
            """).fetchone()[0]
            
            print(f"   ❌ Regra 4: Dias completos sem nenhum registro na série (Lacunas): {r4}")

            df_tendencia = con.execute("""
                SELECT

                    COUNT(*) as qtd_ocorrencias
                FROM v_temporalidade_validada
                WHERE data_validada IS NOT NULL
                GROUP BY ano_mes
                ORDER BY ano_mes;
            """).df()

            # Renderização do Gráfico de Linha Confiável
            plt.figure(figsize=(12, 4.5))
            plt.plot(df_tendencia['ano_mes'], df_tendencia['qtd_ocorrencias'], 
                     marker='o', color='#2b5c8f', linewidth=2, markersize=4)
            
            plt.title('Evolução Mensal do Volume de Acidentes (Série Histórica)', fontsize=12, fontweight='bold')
            plt.xlabel('Linha do Tempo (Ano-Mês)')
            plt.ylabel('Quantidade de Ocorrências')

            ticks_intervalo = max(1, len(df_tendencia) // 12)
            plt.xticks(df_tendencia['ano_mes'][::ticks_intervalo], rotation=45, ha='right')
            
            plt.grid(True, linestyle='--', alpha=0.6)
            plt.tight_layout()
            plt.show()

    else:
        print("⚠️ Erro: A coluna 'data_inversa' não foi localizada no layout da base.")
else:
